query getAllMessages {
  allMessages {
    ...messageFields
  }
}

query getMessageById($searchId: String!) {
  messageById(id: $searchId) {
    ...messageFields
  }
}

query getMessagesByName($searchUser: String!) {
  messagesByUser(user: $searchUser) {
    ...messageFields
  }
}

query getMessagesLength {
  messagesLength
}

mutation addMessage($addMessage: MessageInput) {
  addMessage(input: $addMessage) {
    ...messageFields
  }
}

mutation updateMessage($updateMessage: MessageInput) {
  updateMessage(input: $updateMessage) {
    ...messageFields
  }
}

mutation deleteMessage($deleteMessage: String!) {
  deleteMessage(id: $deleteMessage) {
    ...messageFields
  }
}

fragment messageFields on Message {
  id
  user
  message
  timestamp
}
